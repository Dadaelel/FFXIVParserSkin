<script>
	var oldScript;
	function load_js() {
		var head= document.getElementsByTagName('head')[0];
		if(typeof oldScript != "undefined")
		head.removeChild(oldScript);
		oldScript = document.createElement('script');
		oldScript.src = '../jsondata.js?cachebuster=' + new Date().getTime();
		head.appendChild(oldScript);
		//setTimeout(load_js, 2000);
	}
	load_js();
</script>
